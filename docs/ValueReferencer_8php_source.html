402  public function processReferencesInUpdatedDataset(stdClass $old_dataset, stdClass $new_dataset) {
403  // Cycle through the dataset properties being referenced, check for orphans.
404  foreach ($this->getPropertyList() as $property_id) {
405  if (!isset($old_dataset->{$property_id})) {
406  // The old dataset had no value for this property, thus no references
407  // could be deleted. Safe to skip checking for orphan reference.
408  continue;
409  }

412  }
