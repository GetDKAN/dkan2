391  public function processReferencesInUpdatedDataset(stdClass $old_dataset, stdClass $new_dataset) {
392  // Cycle through the dataset properties being referenced, check for orphans.
393  foreach ($this->getPropertyList() as $property_id) {
394  if (!isset($old_dataset->{$property_id})) {
395  // The old dataset had no value for this property, thus no references
